利用本地缓存Caffeine技术实现重复请求拦截

springboot Caffeine 面向切面

更新日期:   2023-08-02

阅读次数:  

在Java Spring Boot中实现高并发下拦截重复请求功能，可以通过以下步骤进行实现：

一、添加相关的依赖

首先，需要添加相关的依赖。在你的pom.xml文件中添加以下依赖：

<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-aop</artifactId>
</dependency>
<dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-cache</artifactId>
</dependency>
<dependency>
    <groupId>cn.hutool</groupId>
    <artifactId>hutool-all</artifactId>
    <version>5.8.16</version>
</dependency>
<dependency>
    <groupId>com.alibaba</groupId>
    <artifactId>fastjson</artifactId>
    <version>2.0.37</version>
</dependency>
<dependency>
    <groupId>com.github.ben-manes.caffeine</groupId>
    <artifactId>caffeine</artifactId>
</dependency>
<dependency>
    <groupId>com.google.guava</groupId>
    <artifactId>guava</artifactId>
    <version>27.1-jre</version>
</dependency>

二、自定义注解

创建一个自定义注解@RepeatRequestIntercept，用于标记需要拦截重复请求的方法。可以在com.example.demo.annotation包下创建一个RepeatRequestIntercept.java文件，内容如下：

import java.lang.annotation.*;
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
@Documented
public @interface RepeatRequestIntercept {
}

三、创建切面类

创建一个切面类RepeatRequestAspect，用于拦截带有@RepeatRequestIntercept注解的方法。可以在com.example.aspect包下创建一个RepeatRequestAspect.java文件，内容如下：

import java.util.concurrent.TimeUnit;
import javax.servlet.http.HttpServletRequest;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.annotation.Pointcut;
import org.checkerframework.checker.nullness.qual.NonNull;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;
import org.springframework.web.multipart.support.StandardMultipartHttpServletRequest;
import com.github.benmanes.caffeine.cache.Caffeine;
import com.github.benmanes.caffeine.cache.LoadingCache;
import cn.hutool.core.util.HashUtil;
import cn.hutool.core.util.IdUtil;
import cn.hutool.core.util.ObjectUtil;
@Aspect
@Component
public class RepeatRequestAspect {    
    /**
     * 日志工具
     */
    Logger logger = LoggerFactory.getLogger(RepeatRequestAspect.class);    
    public static LoadingCache<String, String> loadingCache = Caffeine.newBuilder()
            .maximumSize(100)
            .expireAfterWrite(1, TimeUnit.MINUTES)
            .build(key -> getUUID(key));   
    private static String getUUID(@NonNull String key) {
        StringBuffer sb = new StringBuffer();       
        Thread t = Thread.currentThread();
        String name = t.getName();
        sb.append(key);
        sb.append("-");
        sb.append(name);
        return sb.toString() ;
    }
    @Pointcut("@annotation(com.example.demo.annotation.RepeatRequestIntercept)")
    public void repeatRequestPointcut() {}
    @Around("repeatRequestPointcut()")
    public Object repeatRequestIntercept(ProceedingJoinPoint joinPoint) throws Throwable {
        String requestId = generateRequestId(joinPoint.getArgs());
        if (loadingCache.getIfPresent(requestId) != null) {
            logger.error("重复请求");
//            throw new Exception("请等10秒钟再试");
            return "请等10秒钟再试";
        } else {
            putValue(requestId);
        }
        try {
            // 执行被拦截的方法
            return joinPoint.proceed();
        } finally {
            // 请求处理完成后，移除请求记录
//             loadingCache.synchronous().invalidate(requestId);
        }
    }
    private String putValue(String requestId) throws Exception {
        synchronized (loadingCache) {
            String fastUUID = getUUID(requestId);
            String str_val = loadingCache.get(requestId);
            if(fastUUID.equals(str_val)) {
                logger.info("loadingCache    put  ======>{} ,{}",requestId,str_val);
                loadingCache.put(requestId,fastUUID);
//                loadingCache.put(requestId,val);
            }else {
                logger.error("重复请求");
//                throw new Exception("请等10秒钟再试");
                return "请等10秒钟再试";
            }
            return str_val;
        }
    }
    private String generateRequestId(Object[] args) {
        // 根据请求参数生成唯一的请求ID，可以根据具体场景进行定制
        // 例如可以将请求参数序列化为JSON字符串，然后进行MD5或SHA1计算
        try {
            for (int i = 0; i < args.length; i++) {    
                if(args[i] instanceof StandardMultipartHttpServletRequest) {
                    HttpServletRequest req = (HttpServletRequest)args[i];
                    final String logisticsInterface = req.getParameter("logistics_interface");// 消息内容
                     if (ObjectUtil.isEmpty(logisticsInterface)) {
                         continue;
                     }
//                     logger.info("logisticsInterface====>{}",logisticsInterface);
                     long hash  = HashUtil.mixHash(logisticsInterface);
//                     logger.info("hash====>{}",hash);
                     return String.valueOf(hash);
                }
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
        return IdUtil.fastUUID();
    }
}

四、添加注解

在需要进行重复请求拦截的方法上添加@RepeatRequestIntercept注解。例如，在控制器类中的某个方法上添加该注解，如下所示：

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.ResponseBody;
import org.springframework.web.bind.annotation.RestController;
import com.example.demo.annotation.RepeatRequestIntercept;
import cn.hutool.core.util.ObjectUtil;
@RestController
@RequestMapping("/order")
public class MyController {
    /**
     * 日志工具
     */
    Logger logger = LoggerFactory.getLogger(MyController.class);
    @RepeatRequestIntercept
    @RequestMapping("/create")
    @ResponseBody
    public String orderCreate(HttpServletRequest request, HttpServletResponse response) throws Exception {
        // 处理请求的业务逻辑
        // 定义请求唯一编码
        String success = "true";
        String reason = "";
        String waybillNo = "";// 物流运单号
        // --------------------------获取参数-----------------------//
        final String logisticsInterface = request.getParameter("logistics_interface");// 消息内容
//         logger.info("接收到的消息内容logisticsInterface为:{}" , logisticsInterface);
        if (ObjectUtil.isEmpty(logisticsInterface)) {
            success = "false";
            reason = "logistics_interface消息不能为空";
            return returnTbMsg("", success, reason, waybillNo);
        }
        return returnTbMsg("", success, reason, waybillNo);
    }
    /**
     *
     * @param txLogisticID 推送的数据
     * @param success      成功
     * @param reason       原因
     * @param waybillNo    单号
     * @return 返回值
     */
    private String returnTbMsg(String txLogisticID, String success, String reason, String waybillNo) {
        StringBuffer returnMessage = new StringBuffer(1024);
        returnMessage.append("<responses>");
        returnMessage.append("<responseItems>");
        returnMessage.append("<response>");
        if (!ObjectUtil.isEmpty(txLogisticID)) {
            returnMessage.append("<txLogisticID>");
            returnMessage.append(txLogisticID); // 物流订单号
            returnMessage.append("</txLogisticID>");
        }
        if (!ObjectUtil.isEmpty(waybillNo)) {
            returnMessage.append("<waybillNo>");
            returnMessage.append(waybillNo); // 物流订单号
            returnMessage.append("</waybillNo>");
        }
        returnMessage.append("<success>");
        returnMessage.append(success);
        returnMessage.append("</success>");
        returnMessage.append("<reason>");
        returnMessage.append(reason);
        returnMessage.append("</reason>");
        returnMessage.append("</response>");
        returnMessage.append("</responseItems>");
        returnMessage.append("</responses>");
        String info = returnMessage.toString();
//       logger.info("推送给客户的数据:{}", info);
        return info;
    }
}

五、启动应用程序

启动Spring Boot应用程序，然后就可以在高并发场景下测试重复请求拦截功能了。
通过以上步骤，我们可以实现在高并发场景下拦截重复请求的功能。切面类RepeatRequestAspect会在带有@RepeatRequestIntercept注解的方法执行前进行拦截，判断当前请求是否已存在，如果已存在则抛出异常，否则执行方法逻辑。同时，使用loadingCache来存储请求记录。
